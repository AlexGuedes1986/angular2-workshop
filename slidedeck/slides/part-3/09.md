## Creating a custom service

```
@Injectable()
export class DogService {

  dogs: Dog[];

  constructor() {
      this.dogs = [
          new Dog({name: 'Bear';, breed: 'Collie', age: 4}),
          new Dog({name: 'Wren';, breed: 'Terrier', age: 2}),
          new Dog({name: 'Cosmos';, breed: 'Bullog', age: 8})
      ];
  }

  getDogs(): Dog[] {
      return this.dogs;
  }
}
```
